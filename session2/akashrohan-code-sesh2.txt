#include "stm32f1xx.h"

void delay(volatile uint32_t count)
{
    while (count--);
}

int main(void)
{
    // Enable clock for GPIOA and GPIOB
    RCC->APB2ENR |= RCC_APB2ENR_IOPAEN;
    RCC->APB2ENR |= RCC_APB2ENR_IOPBEN;

    // set modes to ports
    GPIOA->CRL &= ~(GPIO_CRL_MODE1 | GPIO_CRL_CNF1);
    GPIOA->CRL |= GPIO_CRL_MODE1_1;
    GPIOB->CRH &= ~(GPIO_CRH_MODE14 | GPIO_CRH_CNF14);
    GPIOB->CRH |= GPIO_CRH_MODE14_1;
    GPIOB->ODR |=GPIO_ODR_ODR14;  // set pin14 to high to toggle later , i.e alternative switching ( initially pin14-1 , pin1-0 when toggled becomes vice versa)
    while (1)
    {
        GPIOA->ODR ^= GPIO_ODR_ODR1;
        GPIOB->ODR ^=GPIO_ODR_ODR14;   // Toggle pin value
        delay(500000);                  // crude delay
    }
}
