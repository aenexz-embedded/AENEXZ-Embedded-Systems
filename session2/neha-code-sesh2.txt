#include "stm32f1xx.h"

void delay(volatile uint32_t count)
{
    while (count--)
    {
        __NOP();
    }
}

int main(void)
{
    
    RCC->APB2ENR |= (1 << 2);   
    RCC->APB2ENR |= (1 << 3);  

    
    GPIOA->CRL &= ~(0xF << 4);
    GPIOA->CRL |=  (0x2 << 4);

    
    GPIOB->CRH &= ~(0xF << 24);
    GPIOB->CRH |=  (0x2 << 24);

    
    while (1)
    {
       
        GPIOA->BSRR = (1 << 1);

     
        GPIOB->BSRR = (1 << (14 + 16));

        delay(500000);

        
        GPIOA->BSRR = (1 << (1 + 16));

        GPIOB->BSRR = (1 << 14);

        delay(500000);
    }
}
