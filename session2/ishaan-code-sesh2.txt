#include "stm32f1xx.h"

void delay(volatile uint32_t count)
{
    while (count--);
}

int main(void)
{
    // 1. Enable clocks for GPIOA and GPIOB
  RCC->APB2ENR |= (1 << 2);
  RCC->APB2ENR |= (1 << 3);

  // 2. Configure PA1 as output push-pull in CRL
  GPIOA->CRL &= ~(0xF << 4);      // clear the 4 bits for pin 1
  GPIOA->CRL |=  (0x2 << 4);      // set mode bits (e.g. output, 2MHz push-pull — check the mode table)

  // 3. Configure PB14 as output push-pull in CRH
  GPIOB->CRH &= ~(0xF << 24);     // clear the 4 bits for pin 14
  GPIOB->CRH |=  (0x2 << 24);     // set mode bits

  // 4. Alternating blink loop
  while (1)
  {
      GPIOA->ODR |= (1 << 1);      // PA1 ON
      GPIOB->ODR &= ~(1 << 14);    // PB14 OFF
      delay(500000);

      GPIOA->ODR &= ~(1 << 1);     // PA1 OFF
      GPIOB->ODR |= (1 << 14);     // PB14 ON
      delay(500000);
  }
}
